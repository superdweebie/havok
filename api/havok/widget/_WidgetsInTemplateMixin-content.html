                            havok/widget/_WidgetsInTemplateMixin

                            Mixin to supplement _TemplatedMixin when template contains widgets

                            Usage

                            _WidgetsInTemplateMixin is a mixin. It is not intended to be used alone.

                            To extend a class with _WidgetsInTemplateMixin use declare. Eg:

                            1. require(['dojo/_base/declare', 'my/parent/class', 'havok/widget/_WidgetsInTemplateMixin'], function(declare, MyParent, _WidgetsInTemplateMixin){
                            2.     var MyChildClass = declare([MyParent, _WidgetsInTemplateMixin], function(){
                            3.         //other child extensions here
                            4.     });

                            6.     var child = new MyChildClass; //create an instance of MyChildClass
                            7. });
